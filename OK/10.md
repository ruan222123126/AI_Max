# ... 原有的 imports ...
from ai_service import generate_market_report  # <--- 确保你导入了 AI 服务函数

class NewsItem(BaseModel):
    id: int

    title: str
    source: str
    url: str
    sentiment_score: float

# --- 新增：获取新闻接口 ---
@app.get("/news/latest", response_model=List[NewsItem])
async def get_latest_news(limit: int = 10):
    """
    获取最新的财经新闻
    """
    async with engine.connect() as conn:
        query = text("""
            SELECT id, published_at, title, source, url, sentiment_score 
            FROM financial_news 
            ORDER BY published_at DESC 
            LIMIT :limit
        """)
        result = await conn.execute(query, {"limit": limit})
        return [
            {
                "id": row.id,
                "published_at": row.published_at,
                "title": row.title,
                "source": row.source,
                "url": row.url,
                "sentiment_score": row.sentiment_score or 0.0
            }
            for row in result
        ]

# --- 新增：AI 分析接口 ---
class AnalysisRequest(BaseModel):
    symbol: str

@app.post("/ai/analyze")
async def analyze_market(request: AnalysisRequest):
    """
    触发 AI 对指定资产进行分析
    """
    try:
        # 调用 ai_service.py 中的逻辑
        # 注意：你需要确保 generate_market_report 接受 engine 参数或能在内部连接数据库
        report = await generate_market_report(request.symbol, engine)
        return {"symbol": request.symbol, "report": report}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

#### **2. 升级前端界面 (`frontend/app/page.tsx`)**

我们将界面重构为三栏布局：**左侧行情，中间图表，右侧新闻与 AI**。

**请替换 `frontend/app/page.tsx` 的全部内容：**

```tsx
"use client";

import { useEffect, useState } from "react";
import { LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, ResponsiveContainer } from 'recharts';
import { ArrowUp, ArrowDown, Activity, Newspaper, Bot, RefreshCw } from "lucide-react";

// 数据类型定义
interface MarketTick {
  time: string;
  symbol: string;
  price: number;
}

interface NewsItem {
  id: number;
  published_at: string;
  title: string;
  source: string;
  url: string;
}

export default function Home() {
  // 状态管理
  const [latestData, setLatestData] = useState<MarketTick[]>([]);
  const [historyData, setHistoryData] = useState<MarketTick[]>([]);
  const [newsData, setNewsData] = useState<NewsItem[]>([]);
  const [selectedSymbol, setSelectedSymbol] = useState("BTC-USD");
  const [aiReport, setAiReport] = useState<string>("");
  const [isAnalyzing, setIsAnalyzing] = useState(false);

  // 1. 获取基础数据 (行情 + 新闻)
  const fetchBaseData = async () => {
    try {
      const [marketRes, newsRes] = await Promise.all([
        fetch("http://localhost:8000/market/latest"),
        fetch("http://localhost:8000/news/latest")
      ]);
      
      setLatestData(await marketRes.json());
      setNewsData(await newsRes.json());
    } catch (e) { console.error(e); }
  };

  // 2. 获取图表历史数据
  const fetchChartData = async () => {
    try {
      const res = await fetch(`http://localhost:8000/market/history/${selectedSymbol}?limit=50`);
      const json = await res.json();
      setHistoryData(json.reverse());
    } catch (e) { console.error(e); }
  };

  // 3. 触发 AI 分析
  const handleAnalyze = async () => {
    setIsAnalyzing(true);
    setAiReport(""); // 清空旧报告
    try {
      const res = await fetch("http://localhost:8000/ai/analyze", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ symbol: selectedSymbol })
      });
      const data = await res.json();
      setAiReport(data.report);
    } catch (e) {
      setAiReport("分析服务暂时不可用，请检查后端 AI 配置。");
    } finally {
      setIsAnalyzing(false);
    }
  };

  // 生命周期
  useEffect(() => {
    fetchBaseData();
    const timer = setInterval(fetchBaseData, 5000); // 5秒刷新行情
    return () => clearInterval(timer);
  }, []);

  useEffect(() => {
    fetchChartData();
    setAiReport(""); // 切换资产时清空报告
  }, [selectedSymbol]);

  return (
    <main className="min-h-screen bg-black text-slate-200 p-6 font-sans">
      {/* 顶栏 */}
      <header className="flex items-center space-x-3 mb-8 pb-4 border-b border-slate-800">
        <Activity className="w-6 h-6 text-emerald-500" />
        <h1 className="text-xl font-bold tracking-wider text-slate-100">
          AI_MAX <span className="text-slate-500 font-normal">Global Intelligence</span>
        </h1>
      </header>

      <div className="grid grid-cols-1 lg:grid-cols-12 gap-6">
        
        {/* 左侧：资产列表 (3列) */}
        <div className="lg:col-span-3 space-y-3 overflow-y-auto max-h-[80vh]">
          <h2 className="text-xs font-semibold text-slate-500 uppercase tracking-widest mb-4">Market Assets</h2>
          {latestData.map((item) => (
            <div 
              key={item.symbol}
              onClick={() => setSelectedSymbol(item.symbol)}
              className={`p-4 rounded-lg cursor-pointer transition-all border ${
                selectedSymbol === item.symbol 
                  ? "bg-slate-800 border-emerald-500/50" 
                  : "bg-slate-900/50 border-slate-800 hover:bg-slate-800"
              }`}
            >
              <div className="flex justify-between items-center">
                <span className="font-bold text-sm">{item.symbol}</span>
                <span className={`text-xs px-2 py-1 rounded ${Math.random() > 0.5 ? 'bg-emerald-500/10 text-emerald-400' : 'bg-rose-500/10 text-rose-400'}`}>
                   {Math.random() > 0.5 ? "+0.4%" : "-0.2%"}
                </span>
              </div>
              <div className="mt-2 text-xl font-mono text-slate-300">
                {item.price.toLocaleString(undefined, { minimumFractionDigits: 2 })}
              </div>
            </div>
          ))}
        </div>

        {/* 中间：图表与 AI 分析 (6列) */}
        <div className="lg:col-span-6 flex flex-col space-y-6">
          {/* 图表区域 */}
          <div className="bg-slate-900/50 border border-slate-800 p-4 rounded-xl h-[400px]">
            <h3 className="text-sm text-slate-400 mb-4 flex items-center">
              {selectedSymbol} Real-time Trend
            </h3>
            <ResponsiveContainer width="100%" height="90%">
              <LineChart data={historyData}>
                <CartesianGrid strokeDasharray="3 3" stroke="#1e293b" />
                <XAxis dataKey="time" tick={false} stroke="#475569" />
                <YAxis domain={['auto', 'auto']} orientation="right" stroke="#475569" fontSize={11} />
                <Tooltip 
                  contentStyle={{ backgroundColor: '#0f172a', border: '1px solid #1e293b' }}
                  labelStyle={{ color: '#94a3b8' }}
                />
                <Line type="monotone" dataKey="price" stroke="#10b981" strokeWidth={2} dot={false} />
              </LineChart>
            </ResponsiveContainer>
          </div>

          {/* AI 分析区域 */}
          <div className="flex-1 bg-slate-900/30 border border-slate-800 rounded-xl p-6 relative overflow-hidden">
             <div className="flex justify-between items-center mb-4">
               <h3 className="text-emerald-400 font-bold flex items-center gap-2">
                 <Bot size={18} /> AI Strategy Insight
               </h3>
               <button 
                 onClick={handleAnalyze}
                 disabled={isAnalyzing}
                 className="text-xs flex items-center gap-1 bg-emerald-600 hover:bg-emerald-500 text-white px-3 py-1.5 rounded-full transition-colors disabled:opacity-50"
               >
                 {isAnalyzing ? <RefreshCw className="animate-spin" size={14} /> : "Generate Report"}
               </button>
             </div>
             
             <div className="prose prose-invert prose-sm max-w-none h-64 overflow-y-auto custom-scrollbar">
               {aiReport ? (
                 <div className="whitespace-pre-wrap leading-relaxed text-slate-300">
                   {aiReport}
                 </div>
               ) : (
                 <div className="text-slate-600 italic flex flex-col items-center justify-center h-full">
                   {isAnalyzing ? "AI Agent is analyzing market data & news..." : "Click 'Generate Report' to ask AI Agent"}
                 </div>
               )}
             </div>
          </div>
        </div>

        {/* 右侧：新闻流 (3列) */}
        <div className="lg:col-span-3 bg-slate-900/20 border-l border-slate-800 p-4 h-[80vh] overflow-y-auto">
          <h2 className="flex items-center gap-2 text-xs font-bold text-slate-500 uppercase tracking-widest mb-6 sticky top-0 bg-black py-2">
            <Newspaper size={14} /> Global Wire
          </h2>
          <div className="space-y-6">
            {newsData.map((news) => (
              <a href={news.url} target="_blank" key={news.id} className="block group">
                <div className="flex items-start gap-3">
                  <div className="mt-1 w-1.5 h-1.5 rounded-full bg-blue-500 group-hover:bg-emerald-400 transition-colors"></div>
                  <div>
                    <h4 className="text-sm text-slate-300 group-hover:text-emerald-400 transition-colors leading-snug">
                      {news.title}
                    </h4>
                    <p className="text-xs text-slate-600 mt-1">
                      {new Date(news.published_at).toLocaleTimeString()} · {news.source}
                    </p>
                  </div>
                </div>
              </a>
            ))}
          </div>
        </div>

      </div>
    </main>
  );
}

```
